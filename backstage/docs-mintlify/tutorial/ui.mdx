# Step 5: Building the UI

Now let's create the React UI for our chatbot using @vertz/ui.

## Create the Chat Component

Create `src/ui/Chat.tsx`:

```typescript
import { useState } from 'react';
import { Button, Input, Card, VStack, HStack, Text } from '@vertz/ui';

interface Message {
  id: string;
  role: 'user' | 'assistant';
  content: string;
  createdAt: Date;
}

interface ChatProps {
  messages: Message[];
  onSendMessage: (content: string) => void;
  isLoading?: boolean;
}

export function Chat({ messages, onSendMessage, isLoading }: ChatProps) {
  const [input, setInput] = useState('');
  
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (input.trim()) {
      onSendMessage(input);
      setInput('');
    }
  };
  
  return (
    <Card>
      <VStack gap="md">
        {/* Messages */}
        <div style={{ height: '400px', overflowY: 'auto' }}>
          {messages.map((message) => (
            <div
              key={message.id}
              style={{
                display: 'flex',
                justifyContent: message.role === 'user' ? 'flex-end' : 'flex-start',
                padding: '8px',
              }}
            >
              <div
                style={{
                  background: message.role === 'user' ? '#7C3AED' : '#F3F4F6',
                  color: message.role === 'user' ? 'white' : 'black',
                  padding: '12px 16px',
                  borderRadius: '12px',
                  maxWidth: '70%',
                }}
              >
                {message.content}
              </div>
            </div>
          ))}
          {isLoading && (
            <div style={{ padding: '8px', color: '#6B7280' }}>
              Thinking...
            </div>
          )}
        </div>
        
        {/* Input */}
        <form onSubmit={handleSubmit}>
          <HStack gap="sm">
            <Input
              value={input}
              onChange={(e) => setInput(e.target.value)}
              placeholder="Type your message..."
              disabled={isLoading}
            />
            <Button type="submit" disabled={isLoading || !input.trim()}>
              Send
            </Button>
          </HStack>
        </form>
      </VStack>
    </Card>
  );
}
```

## Create the Main App

Create `src/ui/App.tsx`:

```typescript
import { useState, useEffect } from 'react';
import { Container, Heading, VStack } from '@vertz/ui';
import { createClient } from '@vertz/server/client';
import { Chat } from './Chat';
import type { AppRouter } from '../server';

const client = createClient<AppRouter>('http://localhost:3000');

export function App() {
  const [conversations, setConversations] = useState<any[]>([]);
  const [currentConversation, setCurrentConversation] = useState<any>(null);
  const [messages, setMessages] = useState<any[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  
  useEffect(() => {
    loadConversations();
  }, []);
  
  const loadConversations = async () => {
    const data = await client.chat.listConversations.query();
    setConversations(data);
  };
  
  const loadConversation = async (id: string) => {
    const data = await client.chat.getConversation.query({ id });
    setCurrentConversation(data);
    setMessages(data.messages);
  };
  
  const sendMessage = async (content: string) => {
    if (!currentConversation) {
      // Create new conversation
      const newConv = await client.chat.createConversation.mutate({});
      setCurrentConversation(newConv);
      await loadConversations();
    }
    
    setIsLoading(true);
    
    // Add user message
    const userMessage = await client.chat.sendMessage.mutate({
      conversationId: currentConversation.id,
      content,
    });
    
    setMessages((prev) => [...prev, userMessage]);
    
    // TODO: Connect to LLM (next step)
    setIsLoading(false);
  };
  
  return (
    <Container>
      <Heading>Chatbot</Heading>
      <VStack gap="lg">
        {/* Sidebar */}
        <div style={{ width: '250px' }}>
          <Button onClick={() => {
            setCurrentConversation(null);
            setMessages([]);
          }}>
            New Chat
          </Button>
          {conversations.map((conv) => (
            <div
              key={conv.id}
              onClick={() => loadConversation(conv.id)}
              style={{ cursor: 'pointer', padding: '8px' }}
            >
              {conv.title}
            </div>
          ))}
        </div>
        
        {/* Chat */}
        <Chat
          messages={messages}
          onSendMessage={sendMessage}
          isLoading={isLoading}
        />
      </VStack>
    </Container>
  );
}
```
